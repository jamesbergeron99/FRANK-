const express = require('express'); const multer = require('multer'); const pdf = require('pdf-parse'); const path = require('path'); const cors = require('cors'); const { GoogleGenerativeAI } = require("@google/generative-ai"); require('dotenv').config(); const app = express(); const PORT = process.env.PORT || 10000; app.use((req, res, next) => { res.setHeader("Content-Security-Policy", "frame-ancestors 'self' *"); res.setHeader("X-Frame-Options", "ALLOWALL"); next(); }); app.use(cors()); app.use(express.json({limit: '100mb'})); app.use(express.static(path.join(__dirname, 'public'))); const upload = multer({ storage: multer.memoryStorage() }); const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY || ""); const FRANK_IDENTITY = `You are Frank, the $5 Forensic Script Doctor. You are a sophisticated, flamboyant, and brutally honest Studio Executive. STRICT OPERATING PROTOCOL (NO DEVIATION): 1. NO ONE-SENTENCE ANSWERS: Every single section, without exception, must be a substantial, multi-sentence narrative paragraph. 2. NARRATIVE FLOW: Do not just list issues. Contextualize the work. Use theatrical, flamboyant industry language. 3. THE TRIPLE THREAT: For every parameter, naturally weave in the Problem, the Consequence, and the Fix Direction into the paragraph. 4. COORDINATE AUDIT: Every observation MUST cite a Page Number and a Direct Quote from the script. 5. TOP 3 ISSUES: Lead the audit with the "TOP 3 ISSUES TO FIX" for immediate priority action. MANDATORY OUTPUT SEQUENCE: - SECTION 1: SPELLING & GRAMMAR AUDIT (Deep, page-specific receipts). - SECTION 2: FORMATTING AUDIT (Executive review of industry standards). - SECTION 3: THE LOG-LINE & SYNOPSIS. - SECTION 4: THE TOP 3 ISSUES (The most critical priorities for the writer). - SECTION 5: 18-POINT FORENSIC AUDIT (18 massive, quote-heavy, narrative paragraphs). - SECTION 6: FINAL VERDICT BLOCK (RECOMMEND, CONSIDER, or PASS).`; app.post('/analyze', upload.array('scripts', 10), async (req, res) => { const mode = req.body.mode || 'Feature Film'; const previousContext = req.body.previousContext || ""; if (!req.files || req.files.length === 0) return res.status(400).json({ message: "No pages, honey." }); try { let fullText = ""; for (const file of req.files) { const data = await pdf(file.buffer); fullText += "\n--- SCRIPT: " + file.originalname + " ---\n" + data.text; } const model = genAI.getGenerativeModel({ model: "gemini-3-flash-preview", systemInstruction: FRANK_IDENTITY }); const prompt = `Mode: ${mode}. Darling, give me the absolute gold standard. I want the full, flamboyant, quote-heavy forensic audit. No short answers. No fluff. Use your deep knowledge of film history. PREVIOUS CONTEXT (MEMORIZE): ${previousContext} NEW SCRIPT PAGES: \n\n ${fullText.substring(0, 100000)}`; const result = await model.generateContent(prompt); res.json({ message: result.response.text() }); } catch (err) { res.status(500).json({ message: "Frank is indisposed, darling." }); } }); app.get('/voice-settings', (req, res) => res.json({ apiKey: process.env.FRANK_VOICE_API_KEY })); app.listen(PORT, '0.0.0.0', () => console.log("Final Production Master Active."));
